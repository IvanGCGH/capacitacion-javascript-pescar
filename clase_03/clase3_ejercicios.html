<body>
    <div class="juego">
        <h1>Juego de Adivinanza</h1>
        <p>Adivina el n√∫mero entre 1 y 10</p>
        
        <input type="number" id="intento" min="1" max="10" placeholder="?">
        <br>
        <button onclick="adivinar()">Adivinar</button>
        <button onclick="reiniciar()">Nuevo Juego</button>
        
        <div id="mensaje"></div>
        <div id="intentos">Intentos: 0</div>
    </div>

    <script>
        let numeroSecreto;
        let cantidadIntentos = 0;

        function iniciarJuego() {
            numeroSecreto = Math.floor(Math.random() * 10) + 1;
            cantidadIntentos = 0;
            document.getElementById('mensaje').textContent = '';
            document.getElementById('mensaje').className = '';
            document.getElementById('intentos').textContent = 'Intentos: 0';
            document.getElementById('intento').value = '';
            document.getElementById('intento').disabled = false;
        }

        function adivinar() {
            const intento = parseInt(document.getElementById('intento').value);
            const mensaje = document.getElementById('mensaje');
            
            if (isNaN(intento) || intento < 1 || intento > 10) {
                mensaje.textContent = 'Por favor ingresa un n√∫mero entre 1 y 10';
                mensaje.className = '';
                return;
            }
            
            cantidadIntentos++;
            document.getElementById('intentos').textContent = `Intentos: ${cantidadIntentos}`;
            
            if (intento === numeroSecreto) {
                mensaje.textContent = `¬°CORRECTO! Lo lograste en ${cantidadIntentos} intentos`;
                mensaje.className = 'correcto';
                document.getElementById('intento').disabled = true;
            } else if (intento < numeroSecreto) {
                mensaje.textContent = 'Muy bajo, intenta con un n√∫mero mayor';
                mensaje.className = 'incorrecto';
            } else {
                mensaje.textContent = 'Muy alto, intenta con un n√∫mero menor';
                mensaje.className = 'incorrecto';
            }
            
            document.getElementById('intento').value = '';
            document.getElementById('intento').focus();
        }

        function reiniciar() {
            iniciarJuego();
        }

        // Permitir adivinar con Enter
        document.getElementById('intento').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                adivinar();
            }
        });

        // Iniciar juego al cargar la p√°gina
        iniciarJuego();
    </script>
